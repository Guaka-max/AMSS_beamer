\definecolor{AcademicBlue}{RGB}{0, 0, 255}

% 计数器:在 \AtBeginSection 钩子里随目录一起重置(见正文)
\newcounter{mycitecounter}

% 正文上标引用: \upcite{key}
\newcommand{\upcite}[1]{%
  \textsuperscript{\textcolor{AcademicBlue}{[\ref{#1}]}}%
}

% 帧底部参考条目: \citeitem{label}{作者}{题目}{期刊}
\newcommand{\citeitem}[4]{%
  \refstepcounter{mycitecounter}\label{#1}%
  {\scriptsize\color{AcademicBlue}%
    [\themycitecounter] #2, ``#3,'' \textit{#4}.}\\[1pt]%
}

%%%=============== 用户配置区(按需修改)========================
\title[Short Title]{Presentation Title}
% \subtitle{Subtitle}

% 方括号里的"短版本"用于底部栏,花括号里的完整版本用于标题页。
% 若不写短版本,底部栏会显示完整内容(含邮箱),容易被截断。
% 长版本里如果含 \\、\medskip 等命令,需用 \texorpdfstring 包一层,

\author[Author]{\texorpdfstring{Author Name \\ \medskip \small author@example.com}{Author Name}}

\institute[Short Inst.]{Institution / Affiliation}

% 注意:\AtBeginSection 是"覆盖式"钩子,只能声明一次,
% 把所有要在 section 开始时执行的事都放在这一个块里。
\AtBeginSection[]{%
  \setcounter{mycitecounter}{0}%
  \begin{frame}
    \frametitle{Outline}
    \tableofcontents[currentsection,hideothersubsections]
  \end{frame}
}

\section{Citation System}

\begin{frame}[allowframebreaks]{In-text citation example}

  本模板提供学术蓝色右上角引用样式,例如 \upcite{ref1} 与 \upcite{ref2}。
  每进入新的 \texttt{\textbackslash section},编号自动重置为 1。

  \vfill
  \begin{minipage}[b]{\textwidth}
    \raggedright
    \rule{0.3\textwidth}{0.3pt}\\[2pt]
    \citeitem{ref1}{A. Author}{Sample paper title}{Journal Name, Year}
    \citeitem{ref2}{B. Author, C. Author}{Another sample title}{Conference, Year}
  \end{minipage}

\end{frame}

\section{Theorems and Blocks}

\begin{frame}{Definitions, theorems and proofs}

  Beamer 内置 \texttt{definition}, \texttt{theorem}, \texttt{lemma},
  \texttt{corollary}, \texttt{proof} 等环境,可直接使用。

  \begin{definition}[Concept name]
    一段定义文字。设 $f : X \to Y$ 为映射,$\dots$
  \end{definition}

  \begin{theorem}[Main result]
    对任意 $x \in \mathbb{R}$,有 $x^2 \geq 0$,等号成立当且仅当 $x = 0$。
  \end{theorem}

  \begin{proof}
    $x^2 = x \cdot x$,实数与自身相乘所得非负。\qedhere
  \end{proof}

\end{frame}

\begin{frame}{Block environments}

  Beamer 提供三种带颜色的内容块,分别用于普通说明、警示和示例。

  \begin{block}{Standard block}
    用于一般性结论、说明或概览。
  \end{block}

  \begin{alertblock}{Alert block}
    用于强调需要注意的重点或限制条件。
  \end{alertblock}

  \begin{exampleblock}{Example block}
    用于举例说明,通常呈现为绿色。
  \end{exampleblock}

\end{frame}

\section{Tables}

% 注意:本 frame 含 \verb,必须加 [fragile] 选项。
\begin{frame}[fragile]{Table I: booktabs three-line table}

  使用 \texttt{booktabs} 的 \verb|\toprule|, \verb|\midrule|,
  \verb|\bottomrule| 制作干净的三线表。

  \begin{table}[h]
    \centering
    \caption{Example three-line table.}
    \begin{tabular}{lcc}
      \toprule
      Item & Value & Unit \\
      \midrule
      A & 1.23 & m \\
      B & 4.56 & s \\
      C & 7.89 & kg \\
      \bottomrule
    \end{tabular}
  \end{table}

\end{frame}

\begin{frame}{Table II: tabularx with wrapping cells}

  当列内容较长、需要按页面宽度自动换行时,使用 \texttt{tabularx}
  环境配合自定义列类型 \texttt{L}(导言区已定义)。

  \begin{table}[h]
    \centering
    \caption{Example table with wrapping description column.}
    \begin{tabularx}{\textwidth}{l L}
      \toprule
      Topic & Description \\
      \midrule
      Item 1 & 较长的描述文字,会根据列宽自动换行,适合放置说明性内容。\\
      Item 2 & 另一段较长的描述,同样会自动换行。\\
      \bottomrule
    \end{tabularx}
  \end{table}

\end{frame}

\section{Figures}

% 注意:本 frame 含 \verb,必须加 [fragile]。
\begin{frame}[fragile]{Single figure with caption}

  用 \texttt{figure} 环境插入图片,\verb|\includegraphics| 设置宽度,
  \verb|\caption{}| 添加标题。下面的 \texttt{example-image-a} 是
  \texttt{mwe} 宏包提供的占位图,所有 TeX Live 发行版自带,适合做
  模板示例,不需要额外上传图片文件。

  \begin{figure}[h]
    \centering
    \includegraphics[width=0.55\textwidth]{example-image-a}
    \caption{An example placeholder image (from \texttt{mwe}).}
  \end{figure}

\end{frame}

\begin{frame}{Two figures side by side}

  用 \texttt{columns} 环境横向排列多个图像。
  \texttt{[T]} 让两列顶部对齐;\texttt{[c]} 则垂直居中。

  \vspace{4pt}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \centering
      \includegraphics[width=\linewidth]{example-image-a}\\[2pt]
      {\small (a) 左图说明}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \includegraphics[width=\linewidth]{example-image-b}\\[2pt]
      {\small (b) 右图说明}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Figure with surrounding text}

  图文混排:左侧放图,右侧放说明文字或列表。

  \vspace{4pt}
  \begin{columns}[c]
    \begin{column}{0.42\textwidth}
      \centering
      \includegraphics[width=\linewidth]{example-image-c}
    \end{column}
    \begin{column}{0.55\textwidth}
      此处可以是图像的说明:
      \begin{itemize}
        \item 图中展示的关键现象
        \item 主要观测点
        \item 与已有工作的对比
      \end{itemize}
      或继续展开正文段落。
    \end{column}
  \end{columns}

\end{frame}
